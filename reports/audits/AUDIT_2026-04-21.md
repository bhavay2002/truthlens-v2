**Issues:**

**Fix:**

def load_checkpoint(model, *, checkpoint_dir, optimizer=None, scheduler=None,
                    map_location=None, strict=True, allow_missing=()):
    ...
    result = model.load_state_dict(checkpoint["model_state_dict"], strict=False)
    missing = [k for k in result.missing_keys
               if not any(k.startswith(p) for p in allow_missing)]

def _gradient_importance(self, input_ids, attention_mask):
    device = self._resolve_device()
    if device is not None:
        input_ids = input_ids.to(device)
        attention_mask = attention_mask.to(device)
    self.model.eval()
